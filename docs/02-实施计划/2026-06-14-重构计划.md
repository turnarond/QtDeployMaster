```cpp
#include <QVariant>
#include <QString>

class DeployEvent {
public:
    enum Type {
        ConnectionStatusChanged,  // 连接状态变化
        TaskStarted,              // 任务开始
        TaskProgress,             // 任务进度
        TaskFinished,             // 任务完成
        ErrorOccurred,            // 错误发生
        LogMessage,               // 日志消息
        UploadRequest,            // 上传请求
        CommandExecute            // 命令执行请求
    };

    DeployEvent(Type type = TaskFinished, const QVariant& data = QVariant(), const QString& senderId = "")
        : type(type), data(data), senderId(senderId) {}

    Type type;
    QVariant data;
    QString senderId;
};

Q_DECLARE_METATYPE(DeployEvent)
```

```cpp
class EventBus : public QObject {
    Q_OBJECT

private:
    EventBus(QObject* parent = nullptr) : QObject(parent) {}
    ~EventBus() = default;

    QMap<DeployEvent::Type, QList<QObject*>> m_subscribers;

public:
    static EventBus* instance() {
        static EventBus instance;
        return &instance;
    }

    void subscribe(DeployEvent::Type type, QObject* receiver, const char* slot);
    void unsubscribe(QObject* receiver);
    void postEvent(const DeployEvent& event);

signals:
    void eventPosted(const DeployEvent& event);
};
```

- [ ] **Step 2: 创建 EventBus 实现**

```cpp
#include "EventBus.h"

void EventBus::subscribe(DeployEvent::Type type, QObject* receiver, const char* slot) {
    if (!m_subscribers[type].contains(receiver)) {
        m_subscribers[type].append(receiver);
        connect(this, &EventBus::eventPosted, receiver, slot);
    }
}

void EventBus::unsubscribe(QObject* receiver) {
    for (auto& list : m_subscribers) {
        list.removeAll(receiver);
    }
    disconnect(this, nullptr, receiver, nullptr);
}

void EventBus::postEvent(const DeployEvent& event) {
    emit eventPosted(event);
}
```

```cpp
#include "ConnectionPool.h"

QObject* ConnectionPool::getConnection(const QString& ip, int port, ConnectionType type) {
    QString key = QString("%1:%2:%3").arg(ip).arg(port).arg(static_cast<int>(type));
    
    if (m_connections.contains(key)) {
        return m_connections[key];
    }
    
    return nullptr;
}

void ConnectionPool::releaseConnection(QObject* connection) {
    QString key = m_connections.key(connection);
    if (!key.isEmpty()) {
        m_connections.remove(key);
        connection->deleteLater();
    }
}

void ConnectionPool::setCredentials(const QString& user, const QString& pass) {
    m_user = user;
    m_pass = pass;
}

ConnectionPool::ConnectionStatus ConnectionPool::getStatus(const QString& ip) const {
    for (const QString& key : m_connections.keys()) {
        if (key.startsWith(ip)) {
            return Connected;
        }
    }
    return Disconnected;
}
```

- [ ] **Step 2: 创建 FtpPresenter 实现**

```cpp
#include "FtpPresenter.h"
#include "../model/FtpManager.h"
#include "../framework/EventBus.h"

FtpPresenter::FtpPresenter(QObject* parent) : QObject(parent) {
    m_manager = new FtpManager(this);
    EventBus::instance()->subscribe(DeployEvent::UploadRequest, this, SLOT(onEventPosted(const DeployEvent&)));
    
    connect(m_manager, &FtpManager::uploadProgress, this, &FtpPresenter::onUploadProgress);
    connect(m_manager, &FtpManager::uploadFinished, this, &FtpPresenter::onUploadFinished);
}

FtpPresenter::~FtpPresenter() {
    EventBus::instance()->unsubscribe(this);
}

void FtpPresenter::onEventPosted(const DeployEvent& event) {
    if (event.type == DeployEvent::UploadRequest) {
        AppState::instance()->setIsBusy(true);
        AppState::instance()->setCurrentTask("FTP 上传");
        AppState::instance()->setTaskProgress(0);

        QVariantMap data = event.data.toMap();
        QStringList items = data["items"].toStringList();
        QStringList targets = data["targets"].toStringList();
        QString user = data["user"].toString();
        QString pass = data["pass"].toString();
        QString remotePath = data["remotePath"].toString();

        m_manager->setCredentials(user, pass);
        m_manager->uploadFiles(items, targets, remotePath);
    }
}

void FtpPresenter::onUploadProgress(int progress) {
    AppState::instance()->setTaskProgress(progress);
    EventBus::instance()->postEvent(DeployEvent(DeployEvent::TaskProgress, progress));
}

void FtpPresenter::onUploadFinished(bool success, const QStringList& successes, const QStringList& failures) {
    AppState::instance()->setIsBusy(false);
    AppState::instance()->setTaskProgress(100);

    QVariantMap result;
    result["success"] = success;
    result["successes"] = successes;
    result["failures"] = failures;
    
    EventBus::instance()->postEvent(DeployEvent(DeployEvent::TaskFinished, result));
    
    QString log = QString("FTP上传完成: 成功%1个, 失败%2个").arg(successes.size()).arg(failures.size());
    EventBus::instance()->postEvent(DeployEvent(DeployEvent::LogMessage, log));
}
```

```cpp
#include "FtpManager.h"
#include <QtConcurrent/QtConcurrent>
#include "../framework/ConnectionPool.h"

FtpManager::FtpManager(QObject* parent) : QObject(parent) {}

void FtpManager::setCredentials(const QString& user, const QString& pass) {
    m_user = user;
    m_pass = pass;
    ConnectionPool::instance()->setCredentials(user, pass);
}

void FtpManager::uploadFiles(const QStringList& items, const QStringList& targets, const QString& remotePath) {
    QtConcurrent::run([=]() {
        QStringList successes, failures;
        int totalSteps = items.size() * targets.size();
        int currentStep = 0;

        for (const QString& target : targets) {
            bool deviceSuccess = true;
            
            for (const QString& item : items) {
                // 模拟上传操作
                emit uploadProgress((++currentStep * 100) / totalSteps);
                
                // 实际上传逻辑...
                bool success = true; // 实际应该是上传结果
                
                if (!success) {
                    deviceSuccess = false;
                }
            }
            
            if (deviceSuccess) {
                successes << target;
            } else {
                failures << target;
            }
        }

        emit uploadFinished(failures.isEmpty(), successes, failures);
    });
}
```

```cpp
            AppState::instance()->setIsBusy(false);
            manager->deleteLater();
        });
    }
}
```
